const SS_ID = SpreadsheetApp.getActive().getId(); function doGet(e) { const action = e.parameter.action || ''; if (action === 'ping') { return jsonRes({ok:true, message:'STOCKR API is running'}); } if (action === 'getAll') { return jsonRes(getAll()); } return jsonRes({ok:false, error:'Unknown action'}); } function doPost(e) { let body; try { body = JSON.parse(e.postData.contents); } catch(err) { return jsonRes({ok:false,error:'Invalid JSON'}); } const action = body.action || ''; if (action === 'upsertProduct') return jsonRes(upsertProduct(body)); if (action === 'deleteProduct') return jsonRes(deleteProduct(body)); if (action === 'stockAction') return jsonRes(stockAction(body)); if (action === 'upsertBarcode') return jsonRes(upsertBarcode(body)); if (action === 'deleteBarcode') return jsonRes(deleteBarcode(body)); if (action === 'setAlert') return jsonRes(setAlert(body)); return jsonRes({ok:false, error:'Unknown action'}); } function jsonRes(obj) { return ContentService.createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); } function getSheet(name) { return SpreadsheetApp.openById(SS_ID).getSheetByName(name); } function initSheets() { const ss = SpreadsheetApp.openById(SS_ID); const sheets = { products: ['group_id','name','unit','volume','volume_unit','created_at'], barcodes: ['barcode','group_id','store_name','price','memo','created_at'], inventory: ['group_id','qty','alert_threshold','updated_at'], history: ['timestamp','group_id','product_name','barcode','action','qty','after_qty','store_name','price','user'] }; Object.entries(sheets).forEach(([name, headers]) => { let sh = ss.getSheetByName(name); if (!sh) { sh = ss.insertSheet(name); sh.appendRow(headers); } }); } function getAll() { try { const toArr = sh => { const rows = sh.getDataRange().getValues(); const h = rows[0]; return rows.slice(1).map(r => { const o = {}; h.forEach((k,i) => o[k] = r[i]); return o; }); }; return { ok: true, products: toArr(getSheet('products')), barcodes: toArr(getSheet('barcodes')), inventory: toArr(getSheet('inventory')), }; } catch(e) { return {ok:false, error:e.message}; } } function upsertProduct(b) { try { const sh = getSheet('products'); const rows = sh.getDataRange().getValues(); const h = rows[0]; const gIdx = h.indexOf('group_id'); const existing = rows.slice(1).findIndex(r => r[gIdx] === b.group_id); const now = new Date().toISOString(); const row = [b.group_id, b.name, b.unit||'個', b.volume||'', b.volume_unit||'', now]; if (existing >= 0) { sh.getRange(existing+2, 1, 1, row.length).setValues([row]); } else { sh.appendRow(row); // inventory行も作成 const inv = getSheet('inventory'); const invRows = inv.getDataRange().getValues(); const invGIdx = invRows[0].indexOf('group_id'); const invEx = invRows.slice(1).findIndex(r => r[invGIdx] === b.group_id); if (invEx < 0) inv.appendRow([b.group_id, 0, 0, now]); } return {ok:true}; } catch(e) { return {ok:false, error:e.message}; } } function deleteProduct(b) { try { const gid = b.group_id; ['products','barcodes','inventory'].forEach(name => { const sh = getSheet(name); const rows = sh.getDataRange().getValues(); const gIdx = rows[0].indexOf('group_id'); for (let i = rows.length-1; i >= 1; i--) { if (rows[i][gIdx] === gid) sh.deleteRow(i+1); } }); return {ok:true}; } catch(e) { return {ok:false, error:e.message}; } } function stockAction(b) { try { const move = b.move || ''; // ← 'in' / 'out' / 'set' const gid = b.group_id; const qty = Number(b.qty) || 0; const inv = getSheet('inventory'); const rows = inv.getDataRange().getValues(); const h = rows[0]; const gIdx = h.indexOf('group_id'); const qIdx = h.indexOf('qty'); const rowIdx = rows.slice(1).findIndex(r => r[gIdx] === gid); if (rowIdx < 0) return {ok:false, error:'Product not found in inventory'}; let current = Number(rows[rowIdx+1][qIdx]) || 0; let after; if (move === 'in') after = current + qty; else if (move === 'out') after = Math.max(0, current - qty); else if (move === 'set') after = qty; else return {ok:false, error:'Invalid move: ' + move}; inv.getRange(rowIdx+2, qIdx+1).setValue(after); inv.getRange(rowIdx+2, h.indexOf('updated_at')+1).setValue(new Date().toISOString()); // history const hist = getSheet('history'); hist.appendRow([ new Date().toISOString(), gid, b.product_name||'', b.barcode||'', move, qty, after, b.store_name||'', b.price||'', b.user||'' ]); return {ok:true, afterQty:after}; } catch(e) { return {ok:false, error:e.message}; } } function upsertBarcode(b) { try { const sh = getSheet('barcodes'); const rows = sh.getDataRange().getValues(); const h = rows[0]; const bcIdx = h.indexOf('barcode'); const existing = rows.slice(1).findIndex(r => r[bcIdx] === b.barcode); const now = new Date().toISOString(); const row = [b.barcode, b.group_id, b.store_name||'', b.price||'', b.memo||'', now]; if (existing >= 0) { sh.getRange(existing+2, 1, 1, row.length).setValues([row]); } else { sh.appendRow(row); } return {ok:true}; } catch(e) { return {ok:false, error:e.message}; } } function deleteBarcode(b) { try { const sh = getSheet('barcodes'); const rows = sh.getDataRange().getValues(); const bcIdx = rows[0].indexOf('barcode'); for (let i = rows.length-1; i >= 1; i--) { if (rows[i][bcIdx] === b.barcode) { sh.deleteRow(i+1); break; } } return {ok:true}; } catch(e) { return {ok:false, error:e.message}; } } function setAlert(b) { try { const inv = getSheet('inventory'); const rows = inv.getDataRange().getValues(); const h = rows[0]; const gIdx = h.indexOf('group_id'); const aIdx = h.indexOf('alert_threshold'); const rowIdx = rows.slice(1).findIndex(r => r[gIdx] === b.group_id); if (rowIdx >= 0) { inv.getRange(rowIdx+2, aIdx+1).setValue(Number(b.threshold)||0); } return {ok:true}; } catch(e) { return {ok:false, error:e.message}; } }
